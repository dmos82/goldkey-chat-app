import { DocumentChunk, IDocumentChunk } from '../models/DocumentChunk';
import { cosineSimilarity } from './mathUtils';
 
/**
 * MongoDB store for document chunks
 */
export class MongoStore {
  /**
   * Add document chunks to the store
   */
  static async addDocumentChunks(
    docId: string,
    filename: string,
    chunks: { text: string; embedding: number[]; chunkIndex: number }[]
  ): Promise<void> {
    try {
      const documents = chunks.map((chunk) => ({
        docId,
        metadata: { filename },
        text: chunk.text,
        embedding: chunk.embedding,
        chunkIndex: chunk.chunkIndex,
      }));
 
      await DocumentChunk.insertMany(documents);
    } catch (error) {
      console.error('Error adding document chunks:', error);
      throw error;
    }
  }
 
  /**
   * Find similar chunks based on query embedding
   */
  static async findSimilarChunks(
    queryEmbedding: number[],
    limit: number = 5
  ): Promise<IDocumentChunk[]> {
    try {
      // Get all chunks from MongoDB
      const chunks = await DocumentChunk.find({});
      
      // Calculate similarities and sort
      const chunksWithSimilarity = chunks.map((chunk) => ({
        chunk,
        similarity: cosineSimilarity(queryEmbedding, chunk.embedding),
      }));
 
      return chunksWithSimilarity
        .sort((a, b) => b.similarity - a.similarity)
        .slice(0, limit)
        .map((item) => item.chunk);
    } catch (error) {
      console.error('Error finding similar chunks:', error);
      throw error;
    }
  }
 
  /**
   * Get chunks by document ID
   */
  static async getChunksByDocId(docId: string): Promise<IDocumentChunk[]> {
    try {
      return await DocumentChunk.find({ docId }).sort({ chunkIndex: 1 });
    } catch (error) {
      console.error('Error getting chunks by docId:', error);
      throw error;
    }
  }
 
  /**
   * Delete chunks by document ID
   */
  static async deleteChunksByDocId(docId: string): Promise<void> {
    try {
      await DocumentChunk.deleteMany({ docId });
    } catch (error) {
      console.error('Error deleting chunks by docId:', error);
      throw error;
    }
  }
}
